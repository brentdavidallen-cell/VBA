Creating a NamedGroup and Adding Elements

This example shows how to create a Named Group and how to add elements to the group. 


Sub GroupArcs()
    Dim ee As ElementEnumerator
    Dim el As Element
    Dim arcs As NamedGroupElement

    Set arcs = ActiveModelReference.AddNewNamedGroup("Arcs")

    Set ee = ActiveModelReference.Scan

    Do While ee.MoveNext
        Set el = ee.Current

        'only affect graphic elements
        If el.IsGraphical Then
            'add you tests here
            If el.IsArcElement Then
               arcs.AddMember el
            End If
        End If
    Loop

    '  Do one rewrite at the end since that is much
    '  faster than doing a rewrite every time an element
    '  is added.
    arcs.Rewrite
End Sub
