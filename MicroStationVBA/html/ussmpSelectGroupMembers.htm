Selecting Members of Named Groups

This example shows how to select members of named groups. 


Sub SelectMembers(strPattern As String, model As ModelReference)
    Dim ee As ElementEnumerator

    model.UnselectAllElements
    Set ee = model.ControlElementCache.Scan

    Do While ee.MoveNext
        If ee.Current.IsNamedGroupElement Then
            With ee.Current.AsNamedGroupElement
                If .Name Like strPattern Then
                    .SelectElements msdMemberTraverseManipulate
                End If
            End With
        End If
    Loop
End Sub

Sub RunSelect()
    SelectMembers "*__169*", ActiveModelReference
End Sub
