Redefines a Saved View


Sub RedefineStandardDisplay()
    Dim ele As SavedViewElement

    Set ele = ActiveDesignFile.FindSavedView("StandardDisplay")
    If Not ele Is Nothing Then
        ele.IsLocked = False
        ActiveModelReference.RemoveElement ele
    End If

    Set ele = CreateSavedViewElement(1, "StandardDisplay", "Standard display as of " & Now)
    ActiveModelReference.AddElement ele
End Sub
