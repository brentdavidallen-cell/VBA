Applies a SavedViewElement to a View


Sub ApplyNamedView(strName As String, viewNumber As Long)
    Dim vw As View
    Dim eleSV As SavedViewElement

    Set eleSV = ActiveDesignFile.FindSavedView(strName)  ' Not using optional arguments

    If Not eleSV Is Nothing Then
        Set vw = ActiveDesignFile.Views(viewNumber)
        vw.ApplySavedViewElement eleSV, msdCopyViewPortApplySize
    End If
End Sub
