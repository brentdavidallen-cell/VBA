Changing the AttachName


Sub ChangeToLogical()
    Dim att As Attachment

    For Each att In ActiveModelReference.Attachments
        Dim strAttachName As String
        Dim position As Long

        strAttachName = att.AttachName
        position = InStrRev(strAttachName, "\")
        If position > 0 Then
            strAttachName = "MyLogical:" & Mid(strAttachName, position + 1)
            att.SetAttachNameDeferred strAttachName
            att.Rewrite
        End If
    Next
End Sub
