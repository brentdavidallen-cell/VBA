Changing a View's Camera Parameters


Sub SetCameraClippingDistance(vw As View, Front As Double, Back As Double)
    Dim vc As New ViewCameraParameters

    '  Set the requested properties
    vc.CameraBackClippingDistance = Back
    vc.CameraFrontClippingDistance = Front

    '  Set the others so SetCameraProperties does not compute defaults
    vc.CameraAngle = vw.CameraAngle
    vc.CameraPosition = vw.CameraPosition
    vc.CameraTarget = vw.CameraTarget
    vc.CameraUpVector = vw.CameraUpVector

    vw.SetCameraProperties vc
    vw.Redraw
End Sub
