Illustrates Differences in ProjectPointOnPerpendicular Methods

Demonstrates the differences in the ProjectPointOnPerpendicular methods. These examples draw lines to show the returned points. 


Sub RunLineSample()
    Dim ele As LineElement
    Dim temp As LineElement
    Dim pntProjected As Point3d
    Dim pntStart As Point3d

    Set ele = CreateLineElement2(Nothing, Point3dFromXY(0, 5), Point3dFromXY(5, 5))
    ele.Color = 3
    ActiveModelReference.AddElement ele
    ele.Redraw

    pntStart = Point3dFromXY(7, 0)

    pntProjected = ele.ProjectPointOnPerpendicular(pntStart, Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 0
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular2(pntStart, Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 1
    ActiveModelReference.AddElement temp
    temp.Redraw
End Sub
Sub RunShapeSample()
    Dim pnts(0 To 2) As Point3d
    Dim ele As ShapeElement
    Dim temp As LineElement
    Dim pntProjected As Point3d
    Dim pntStart As Point3d

    pnts(0) = Point3dFromXY(0, 0)
    pnts(1) = Point3dFromXY(7, 0)
    pnts(2) = Point3dFromXY(3.5, 10)
    Set ele = CreateShapeElement1(Nothing, pnts)
    ele.Color = 3
    ActiveModelReference.AddElement ele
    ele.Redraw

    pntStart = Point3dFromXY(3.5, 5)

    pntProjected = ele.ProjectPointOnPerpendicular(pntStart, Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 0
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular1(pntStart, Point3dFromXY(0, 1), Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 1
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular1(pntStart, Point3dFromXY(12, 3), Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 2
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular1(pntStart, Point3dFromXY(2, -1), Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 4
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntStart = Point3dFromXY(-2, -2)

    pntProjected = ele.ProjectPointOnPerpendicular(pntStart, Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 0
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular1(pntStart, Point3dFromXY(12, 3), Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 1
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular2(pntStart, Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 2
    ActiveModelReference.AddElement temp
    temp.Redraw
End Sub
Sub RunEllipseSamples()
    Dim ele As EllipseElement
    Dim temp As LineElement
    Dim pntProjected As Point3d
    Dim pntStart As Point3d

    Set ele = CreateEllipseElement2(Nothing, Point3dFromXY(4, 4), 3, 3, Matrix3dIdentity)
    ele.Color = 3
    ActiveModelReference.AddElement ele
    ele.Redraw

    pntStart = Point3dFromXY(14, 4)

    pntProjected = ele.ProjectPointOnPerpendicular(pntStart, Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 0
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular1(pntStart, Point3dAdd(ele.CenterPoint, Point3dFromXY(1, 0)), Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 1
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular1(pntStart, Point3dAdd(ele.CenterPoint, Point3dFromXY(-1, 0)), Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 2
    ActiveModelReference.AddElement temp
    temp.Redraw

    pntProjected = ele.ProjectPointOnPerpendicular2(pntStart, Matrix3dIdentity)
    Set temp = CreateLineElement2(Nothing, pntStart, pntProjected)
    temp.Color = 4
    ActiveModelReference.AddElement temp
    temp.Redraw
End Sub
