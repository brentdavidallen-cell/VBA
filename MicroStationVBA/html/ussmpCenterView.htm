Centers the View at the Current Cursor Location


Sub CenterViewAtCurrentPoint()
    Dim vw As View
    Dim ci As CursorInformation

    Set ci = CursorInformation
    Set vw = ci.CurrentView
    '
    '  Set the current view's center to the cursor location
    '
    vw.Center = ci.CurrentPointRaw
    '
    '  Update the View
    '
    vw.Redraw
End Sub
'
'  Utility for ShowCursorInfo
'
Sub ShowXY(prefix As String, pnt As Point3d)
    Debug.Print prefix & "X = " & pnt.X & ", Y = " & pnt.Y
End Sub
'
'  Displays the state of the cursor
'
Sub ShowCursorInfo()
    Dim vw As View

    With CursorInformation
        ShowXY "DP ADJ  ", .DataPointAdjusted
        ShowXY "DP RAW  ", .DataPointRaw
        ShowXY "CP RAW  ", .CurrentPointRaw
        Debug.Print "DP in View " & .DataPointView.Index
        Debug.Print "Cursor in view " & .CurrentView.Index
        Debug.Print "Precision DataPoint = " & .IsDataPointPrecision
        Debug.Print "Cursor in view = " & .IsCurrentlyInView
    End With
End Sub
