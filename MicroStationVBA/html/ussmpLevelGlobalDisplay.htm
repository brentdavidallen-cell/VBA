Changing Level Display for an Attachment


Sub ToggleLevelDisplay(att As Attachment, strLevelPattern As String)
    Dim lvls As Levels
    Dim lvl As Level

    Set lvls = att.Levels
    For Each lvl In att.Levels
        If lvl.Name Like strLevelPattern Then
            lvl.IsDisplayed = Not lvl.IsDisplayed
        End If
    Next

    lvls.Rewrite
    RedrawAllViews
End Sub
