Deleting a Model

This examples illustrates deleting a model. 


Sub DeleteModel(dgn As DesignFile, strName As String)
    Dim index As Integer

    For index = dgn.Models.Count To 1 Step -1
        Dim mdl As ModelReference

        Set mdl = dgn.Models(index)
        If mdl.Name Like strName Then
            dgn.Models.Delete mdl
        End If
    Next
End Sub
